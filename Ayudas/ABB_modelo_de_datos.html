# Paso 2: Función para agregar columnas faltantes con NA
add_missing_columns <- function(df, all_names) {
  missing_cols <- setdiff(all_names, names(df))
  for (col in missing_cols) {
    df[[col]] <- NA
  }
  # Reordenar columnas para mantener el mismo orden
  df[all_names]
}

# Paso 3: Aplicar la función a todos los elementos de la lista
listing_list <- lapply(listing_list, add_missing_columns, all_names = all_column_names)

# Verificación
